\textbf{Integrali}
\begin{center}
	\begin{small}
		\begin{enumerate}
			\item \begin{math}
				      \int x^a\,dx =
				      \Bigg\{\begin{tabular}{ccc}
					      $\frac{x^{a+1}}{a+1} + C$  & $a \neq -1$ & \\
					      $ \ln{\left|x\right|} + C$ & $a = -1$    & \\
				      \end{tabular}
			      \end{math}
			\item \begin{math}
				      \int \ln {x}\,dx = x \ln {x} - x + C
			      \end{math}
			\item \begin{math}
				      \int \frac {1}{\sqrt{x}}\,dx=2\sqrt{x} + C
			      \end{math}
			\item \begin{math}
				      \int e^x\,dx = e^x + C
			      \end{math}
			\item \begin{math}
				      \int a^x\,dx = \frac{a^x}{\ln{a}} + C
			      \end{math}
			\item \begin{math}
				      \int \cos({ax}) \, dx = { \frac{\sin (ax)}{a} } + C
			      \end{math}
			\item \begin{math}
				      \int \sin({ax}) \, dx = { \frac{-\cos(ax)}{a} } + C
			      \end{math}
			\item \begin{math}
				      \int \tan{x} \, dx = -\ln{\left| \cos {x} \right|} + C
			      \end{math}
			\item \begin{math}
				      \int \frac{dx}{\cos^2 x}=\int \sec^2 x \, dx = \tan x + C
			      \end{math}
			\item \begin{math}
				      \int \frac{dx}{\sin^2 x}=\int \csc^2 x \, dx = -\cot x + C
			      \end{math}
			\item \begin{math}
				      \int {\frac{1}{\sqrt{1-x^2}}} \, dx = \arcsin {x} + C
			      \end{math}
			\item \begin{math}
				      \int \frac{dx}{ax + b} = \frac{1}{a} \ln |ax + b| + C
			      \end{math}
			\item \begin{math}
				      \int \frac{1}{x^2 + 1} \, dx = \arctan x + C
			      \end{math}
			\item \begin{math}
				      \int \frac{dx}{x^2 + a^2} = \frac{1}{a} \arctan \frac{x}{a} + C
			      \end{math}
			\item \begin{math}
				      \int \frac{f'(x)}{f(x)} \, dx = \ln|f(x)| + C
			      \end{math}
			\item \textbf{Zamenjava spremenljivke:} Če $u = g(x)$ in $du = g'(x)dx$, potem velja:
			      \begin{math}
				      \int f(g(x))g'(x)\,dx = \int f(u)\,du
			      \end{math}
			\item \textbf{Metoda per partes:} Če $u = u(x)$ in $v' = v'(x)$, potem velja:
			      \begin{math}
				      \int u\,v'\,dx = uv - \int u'\,v\,dx
			      \end{math}
		\end{enumerate}
	\end{small}
\end{center}

\textbf{Logaritmi}
\begin{center}
	\begin{small}
		\begin{enumerate}
			\item \textbf{Definicija:} $\log_a x = y \iff a^y = x$ \quad ($a>0,\ a\neq 1,\ x>0$)
			\item $\log_a 1 = 0$
			\item $\log_a a = 1$
			\item $\log_a(a^x)=x$ \quad in \quad $a^{\log_a x}=x$
			\item $\log_a(xy)=\log_a x+\log_a y$
			\item $\log_a\!\left(\frac{x}{y}\right)=\log_a x-\log_a y$
			\item $\log_a(x^r)=r\log_a x$ \quad ($x>0$)
			\item $\log_a\!\left(\sqrt[n]{x}\right)=\log_a\!\left(x^{1/n}\right)=\frac{1}{n}\log_a x$
			\item \textbf{Menjava osnove:} $\log_a x=\frac{\log_b x}{\log_b a}=\frac{\ln x}{\ln a}$
			\item $\ln x = \log_e x$
			\item $\log_a b=\frac{1}{\log_b a}$
			\item \textbf{Monotonost:} če $a>1$, je $\log_a x$ naraščajoča; če $0<a<1$, je padajoča.
		\end{enumerate}
	\end{small}
\end{center}

\textbf{Eksponentna funkcija}
\begin{center}
	\begin{small}
		\begin{enumerate}
			\item \textbf{Definicija:} $e^x = \exp(x)$ \quad ($e \approx 2.71828$)
			\item $e^0 = 1$, \quad $e^x>0$ za vsak $x\in\mathbb{R}$
			\item \textbf{Pravila potenc:}
			      \[
				      e^{x+y}=e^x e^y,\quad e^{x-y}=\frac{e^x}{e^y},\quad e^{kx}=(e^x)^k
			      \]
			\item \textbf{Inverz:} $\ln(e^x)=x$ in $e^{\ln x}=x$ \quad ($x>0$)
			\item \textbf{Monotonost:} $e^x$ je strogo naraščajoča na $\mathbb{R}$
		\end{enumerate}
	\end{small}
\end{center}
