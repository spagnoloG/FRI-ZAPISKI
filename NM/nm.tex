\section{Linear Systems}

\subsection{Vector Norms}
Vector norm $\|\cdot\|:\R^n \to \R$.
Properties:
\begin{itemize}
  \item PD: $\|x\| \geq 0$, $\|x\| = 0 \Longleftrightarrow x=0$
  \item Homogeneity: $\|\alpha x\| = |\alpha| \cdot \|x\|$
  \item Triangle inequality: $\|x+y\| \leq \|x\|+\|y\|$
\end{itemize}

\begin{tiny}
\[
  \begin{gathered}
    \|x\|_p = \left(\sum_{i=1}^n |x_i|^p\right)^{1/p}, \\
    \|x\|_\infty = \max |x_i|
  \end{gathered}
\]
\end{tiny}

\subsection{Matrix Norms}
Matrix norm $\|\cdot\|:\R^{n\times n}\to\R$.
Properties:
\begin{itemize}
  \item PD: $\|A\|\geq 0$, $\|A\|=0 \Longleftrightarrow A=0$
  \item Homogeneity: $\|\alpha A\|=|\alpha|\cdot\|A\|$
  \item Triangle inequality: $\|A+B\|\leq\|A\|+\|B\|$
  \item Submultiplicativity: $\|AB\|\leq\|A\|\|B\|$
\end{itemize}

\begin{tiny}
\[
  \begin{gathered}
    \|A\|_1=\max_j\sum_{i=1}^n|a_{ij}|, \\
    \|A\|_2=\sqrt{\lambda_{\max}(A^T A)}, \\
    \|A\|_F=\sqrt{\sum_{i,j=1}^n|a_{ij}|^2}
      =\sqrt{\sum_i\sigma_i^2}
      =\sqrt{\operatorname{tr}(A^TA)}, \\
    \|A\|_\infty=\max_i\sum_{j=1}^n|a_{ij}|
  \end{gathered}
\]
\end{tiny}

\subsection{Condition Number / Sensitivity:} 
\[
  \kappa(A)=\|A\|\|A^{-1}\|.
\]
$\kappa(A)$ measures sensitivity: how much relative perturbations/noise in
$A$ or $b$ can change the solution $x$. If $\kappa(A)\approx 1$, the problem is
well-conditioned; if $\kappa(A)\gg 1$, errors are amplified and the matrix is
nearly singular/ill-conditioned.

\subsection{\texorpdfstring{Gaussian Elimination ($\sim\frac{2}{3}n^3$):}
  {Gaussian Elimination (2/3 n3):}}
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
\texttt{for} $k=1,\ldots,n-1$ \texttt{// pivot el.} \\
\> \texttt{for} $i=k+1,\ldots,n$ \texttt{// rows} \\
\> \> $\texttt{xmult}=a_{ik}/a_{kk}$ \\
\> \> $a_{ik}=0$ \\
\> \> \texttt{for} $j=k+1,\ldots,n$ \texttt{// columns}\\
\> \> \> $a_{ij}=a_{ij}-\texttt{xmult}\,a_{kj}$ \\
\> \> \texttt{end} \\
\> \> $b_i=b_i-\texttt{xmult}\,b_k$ \\
\> \texttt{end} \\
\texttt{end}
\end{tabbing}
\end{tiny}

\textbf{Back Substitution} ($\sim n^2$):
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
$x_n=b_n/u_{nn}$ \\
\texttt{for} $i=n-1,\ldots,1$ \\
\> $s=b_i$ \\
\> \texttt{for} $j=i+1,\ldots,n$ \\
\> \> $s=s-u_{ij}x_j$ \\
\> \texttt{end} \\
\> $x_i=s/u_{ii}$ \\
\texttt{end // solving Ux=b}
\end{tabbing}
\end{tiny}

\subsection{LU Decomposition}
Decomposition of $A \in \R^{n\times n}$ into a lower $\Delta$ and an upper $\Delta$ matrix:
\[
  A = LU
\]
start with $L=I$, $U=A$.
\textbf{Algorithm:}
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
\texttt{for} $k=1,\ldots,n-1$ \\
\> \texttt{for} $i=k+1,\ldots,n$ \\
\> \> $l_{ik}=a_{ik}/a_{kk}$ \\
\> \> \texttt{for} $j=k+1,\ldots,n$ \\
\> \> \> $a_{ij}=a_{ij}-l_{ik}a_{kj}$ \\
\> \> \texttt{end} \\
\> \texttt{end} \\
\texttt{end // same as G.E., but store multipliers in L}
\end{tabbing}
\end{tiny}


\textbf{Solving the system $Ax=b$:}
\begin{align*}
  A &= LU                 &&(\sim \tfrac{2}{3}n^3) \\
  Ly &= b                 &&(\sim n^2-n) \\
  Ux &= y                 &&(\sim n^2)
\end{align*}
\subsection{Pivoting}
\[
  PA = LU
\]
\textbf{Partial pivoting:} replace the $j$-th row with the row having the largest
$|a_{ij}|$, $i=j,\ldots,n$.
Used to avoid zero/small pivots: if $a_{jj}$ is tiny, multipliers
$l_{ij}=a_{ij}/a_{jj}$ become large and rounding errors are amplified.
\textbf{Pivot growth:}
\[
  \rho(A)=\frac{\max_{i,j}|\hat u_{ij}|}{\max_{i,j}|a_{ij}|}.
\]
Measures how much entries grow during elimination; small $\rho(A)$ means more
stable elimination. For partial pivoting, $\rho(A)\leq 2^{n-1}$.

\textbf{Permutation matrix:} $P^T=P^{-1}$.
\[
  P = I_{\text{column }i \leftrightarrow j},
  \quad PA = A_{\text{row }i \leftrightarrow j}
\]
Row-swap matrix for swapping rows $j$ and $k$:
\[
  P_{jk}=I_n-E_{jj}-E_{kk}+E_{jk}+E_{kj}.
\]
$P_{jk}A$ swaps
rows $j$ and $k$ of $A$.
Swap the 2nd and 3rd rows of matrix $A$:
\[
  P_{23}A,\qquad
  P_{23} = \begin{bmatrix}
        1 & 0 & 0 \\
        0 & 0 & 1 \\
        0 & 1 & 0
      \end{bmatrix}
\]
\textbf{Solving the system $Ax=b$:}
\begin{align*}
  A &= P^T LU             &&(\sim \tfrac{2}{3}n^3) \\
  Ly &= Pb                &&(\sim n^2) \\
  Ux &= y                 &&(\sim n^2)
\end{align*}
\subsection{\texorpdfstring{Cholesky ($\sim\frac{1}{3}n^3$).}
  {Cholesky (1/3 n3).}}
LU decomposition for PD matrices:
\[
  A = LL^T
\]
\textbf{Algorithm:}
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
\texttt{for} $k=1,\ldots,n$ \texttt{// go along diagonal}\\
\> $l_{kk}=\sqrt{a_{kk}-\sum_{i=1}^{k-1}l_{ki}^2}$ \\
\> \texttt{for} $j=k+1,\ldots,n$ \texttt{// for each el. j below diag.} \\
\> \> \texttt{for} $i=1,\ldots,k-1$ \\
\> \> \> $a_{jk}=a_{jk}-l_{ji}l_{ki}$ \\
\> \> \texttt{end} \\
\> \> $l_{jk}=a_{jk}/l_{kk}$ \\
\> \texttt{end} \\
\texttt{end}
\end{tabbing}
\end{tiny}
Exists when matrix $A$ is:
\begin{itemize}
  \item Symmetric: $A=A^T$
  \item PD: $x^T Ax>0 \quad \forall x \neq 0$
\end{itemize}

\textbf{Strict row diagonal dominance:} $|a_{ii}|>\sum_{j\neq i}|a_{ij}|$ for every $i$.

\subsection{Iterative Methods}
For $Ax=b$, direct methods compute the exact solution $x^*$ up to rounding.
Iterative methods produce approximations $x^{(k)}$ and stop when the required
tolerance is reached:
\[
  \|\hat x-x^*\|\leq \varepsilon
  \quad\text{or usually}\quad
  \|Ax^{(k)}-b\|\leq \varepsilon.
\]
General form:
\[
  x^{(k+1)}=Rx^{(k)}+c.
\]
Useful for large sparse matrices: cheaper memory/work per step, and we can stop
as soon as the approximation is accurate enough.

\subsection{Spectral Radius}
\[
  \rho(R)=\max_i|\lambda_i(R)|.
\]
For $x^{(k+1)}=Rx^{(k)}+c$, convergence for every $x^{(0)}$
$\Longleftrightarrow \rho(R)<1$. Smaller $\rho(R)$ means faster convergence:
the error is roughly multiplied by $\rho(R)$ each step.
\[
  e^{(k)}\approx R^ke^{(0)},\qquad
  \|e^{(k)}\|\sim \rho(R)^k\|e^{(0)}\|.
\]
Convergence rate:
\[
  r(R)=-\log_{10}\rho(R).
\]
For error about $10^{-p}$, approximate number of steps:
\[
  m\approx \frac{p}{r(R)}.
\]

\subsection{\texorpdfstring{Jacobi Iteration ($O(mn)$).}
  {Jacobi Iteration (O(mn)).}}
\[
  x_i^{(k)}=\frac{1}{a_{ii}}
  \left(b_i-\sum_{\substack{j=1\\j\neq i}}^n a_{ij}x_j^{(k-1)}\right),
  \quad i=1,\ldots,n.
\]

\begin{tiny}
\[
x_n^{(k)} = \frac{b_n - a_{n,1}x_1^{(k-1)} - \cdots - a_{n,n-1}x_{n-1}^{(k-1)}}{a_{nn}}
\]
\end{tiny}

\subsection{Gauss-Seidel}

Component form:
\[
  x_i^{(k+1)} = \frac{1}{a_{ii}}
    \left(b_i-\sum_{j<i}a_{ij}x_j^{(k+1)}
    -\sum_{j>i}a_{ij}x_j^{(k)}\right)
\]
More memory-efficient than Jacobi: updates $x_i$ in-place, so it does not need
two separate vectors for old/new iterates.

\subsection{SOR}
Relaxed GS iteration with $\omega\in\R$:
\begin{tiny}
\[
  x_i^{(k)}=(1-\omega)x_i^{(k-1)}+\omega x_i^{\mathrm{GS}}.
\]
\[
  x_i^{(k)}=(1-\omega)x_i^{(k-1)}
  +\frac{\omega}{a_{ii}}\left(b_i-\sum_{j<i}a_{ij}x_j^{(k)}
  -\sum_{j>i}a_{ij}x_j^{(k-1)}\right).
\]
\end{tiny}
If $A$ is PD and $0<\omega<2$, SOR converges.

\subsection{Vectors are conjugate}, if $x^TAy = 0$ ($\langle x,y \rangle_A = x^TAy$).

\subsection{Overdetermined Systems}
more equations than unknowns:
\[
  A \in \R^{n\times m}, \quad b \in \R^n, \quad n\gg m
\]

\subsection{Least Squares Method}
\[
  \tilde x = \operatorname*{argmin}_x \|Ax-b\|
\]

The least-squares solution of $Ax=b$ is $x$, which solves the system $A^T A x = A^T b$.
Using Cholesky:
\begin{align*}
  B &= A^T A,\quad c=A^T b &&(\sim nm^2) \\
  B &= VV^T                &&(\sim \tfrac{1}{3}m^3) \\
  Vy &= c                  &&(\sim m^2) \\
  V^T\tilde x &= y         &&(\sim m^2)
\end{align*}

\subsection{QR Decomposition}
For $A \in \R^{n\times m}$, $n\geq m$
\[
  A=QR
\]
\[
 \quad Q^T=Q^{-1}, \quad R \text{ upper } \triangle
\]
Solve:
\[
  Rx=Q^T b
\]
For least squares, QR is usually better than normal equations
$A^TAx=A^Tb$ because forming $A^TA$ squares the condition number:
\[
  \kappa(A^TA)=\kappa(A)^2.
\]
So if $A$ is already poorly conditioned, $A^TA$ is much worse; QR works with
$A$ directly.
\subsection{Gram--Schmidt}
Orthogonalization of the set
$\{a_1,\ldots,a_n\} \Rightarrow \{q_1,\ldots,q_n\}$, such that
$L(q_i)=L(a_i)$
\[
  q_1=\frac{a_1}{\|a_1\|}, \quad
  q_k=\frac{a_k-\sum_{j=1}^{k-1}\langle a_k,q_j\rangle q_j}{\|q_k\|}
\]
\subsection{Householder Reflection}
\[
  H_w=I-2\frac{ww^T}{w^T w}, \quad H_w^T=H_w, \quad H_w^2=I
\]
\[
  w=x+\sgn(x_1)\|x\|_2\cdot e_1
\]
\[
  R=H_k\ldots H_1 A, \quad Q=H_1\ldots H_k
\]
Geometric meaning: $H_w$ reflects a vector across the hyperplane orthogonal to
$w$. The component parallel to $w$ changes sign, while the perpendicular part is
unchanged:
\[
  x=\alpha w+u,\ u\perp w \quad\Rightarrow\quad H_wx=-\alpha w+u.
\]
Use Householder for QR when numerical stability matters: classical GS can lose
orthogonality if columns are nearly linearly dependent. Householder is not
necessarily fewer flops than GS, but is more stable and standard for dense QR.

\subsection{Singular Value Decomposition (SVD).}
For $A\in\R^{n\times m}$, there exist orthogonal
$U\in\R^{n\times n}$, $V\in\R^{m\times m}$ and diagonal
$\Sigma\in\R^{n\times m}$:
\begin{tiny}
\[
  UU^T=I_n,\quad V^TV=I_m,\quad
  \Sigma=\operatorname{diag}(\sigma_1,\ldots,\sigma_{\min(n,m)})
\]
\[
  \sigma_1\geq\cdots\geq\sigma_{\min(n,m)}\geq0,\qquad
  A=U\Sigma V^T.
\]
\end{tiny}
Right $v_i\in\R^m$ are input directions, and left $u_i\in\R^n$ are output directions:
$Av_i=\sigma_i u_i$.
Singular values are not eigenvalues of $A$ in general:
\[
  \sigma_i=\sqrt{\lambda_i(A^TA)}\geq0.
\]
Eigenvalues solve $Av=\lambda v$ (same input/output direction, square $A$);
singular values measure stretching $Av_i=\sigma_i u_i$ and also exist for
rectangular matrices.

\subsection{Moore--Penrose Inverse}
For $A\in\R^{n\times m}$, the pseudoinverse $A^+\in\R^{m\times n}$ satisfies:
\begin{tiny}
\[
  \begin{gathered}
    AA^+A=A,\quad A^+AA^+=A^+,\\
    (AA^+)^T=AA^+,\quad (A^+A)^T=A^+A.
  \end{gathered}
\]
\end{tiny}
If $A=U\Sigma V^T$ is SVD, then
\[
  A^+=V\Sigma^+U^T,\qquad
  \sigma_i^+=
  \begin{cases}
    1/\sigma_i, & \sigma_i\neq0,\\
    0, & \sigma_i=0.
  \end{cases}
\]
Least squares solution can be written $x=A^+b$.

\section{Finding Eigenvalues}

\subsection{Jordan and Schur Forms}
Eigenvalue methods try to transform $A$ into a simpler similar matrix, because
similar matrices have the same eigenvalues.

\textbf{Jordan form} is the theoretical canonical form:
\[
  A=SJS^{-1}, \qquad J=\operatorname{diag}(J_1,\ldots,J_r).
\]
Each Jordan block has $\lambda$ on the diagonal and $1$ above it:
\begin{tiny}
\[
  J_k=
  \begin{bmatrix}
    \lambda & 1       &        & 0\\
            & \lambda & \ddots & \\
            &         & \ddots & 1\\
    0       &         &        & \lambda
  \end{bmatrix}.
\]
\end{tiny}
If $A$ has enough independent eigenvectors, then $J$ is diagonal. Otherwise,
Jordan blocks appear. Computing Jordan form is numerically unstable, so it is
mostly theoretical.

\textbf{Schur form} is the practical replacement for Jordan form:
\[
  Q^*AQ=T,
\]
where $Q$ is unitary and $T$ is upper triangular. The eigenvalues are the
diagonal entries of $T$.

For real matrices:
\[
  U^TAU=T,
\]
where $U$ is orthogonal and $T$ is quasi-upper triangular: it has $1\times1$
blocks for real eigenvalues and $2\times2$ blocks for complex conjugate pairs.

QR iteration moves the matrix toward Schur form using similarity transformations:
\[
  A_i=Q_iR_i,\qquad A_{i+1}=R_iQ_i=Q_i^TA_iQ_i.
\]
So eigenvalues stay the same, while $A_i$ becomes more triangular after many
iterations.


\subsection{Power Method}
Dominant eigenpair. Input: $A\in\R^{n\times n}$, $x_0\neq0$, tol.
Output: $(x,\lambda)$, where $|\lambda|$ is largest.
If $x_0$ is already an eigenvector and $\|x_0\|_2=1$, then
\[
  Ax_0=\lambda x_0
  \quad\Rightarrow\quad
  x_0^TAx_0=\lambda x_0^Tx_0=\lambda.
\]
So estimate eigenvalues by the Rayleigh quotient $\tilde\lambda=x^TAx$ for
normalized $x$.
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\kill
$x_0=x_0/\|x_0\|_2,\quad \tilde\lambda_0=x_0^TAx_0,\quad i=0$ \\
\texttt{while} $\|Ax_i-\tilde\lambda_i x_i\|>\texttt{tol}$ \\
\> $y_{i+1}=Ax_i$ \\
\> $x_{i+1}=y_{i+1}/\|y_{i+1}\|_2$ \\
\> $\tilde\lambda_{i+1}=x_{i+1}^TAx_{i+1}$ \\
\> $i=i+1$ \\
\texttt{end}
\end{tabbing}
\end{tiny}
Each multiplication by $A$ amplifies the dominant eigenvector direction.
Convergence rate $\approx |\lambda_2/\lambda_1|^k$.

\subsection{Inverse Iteration}
For a shift $\sigma\approx\lambda$, use the power method on
$(A-\sigma I)^{-1}$; obtain the eigenvalue closest to $\sigma$.
Input: $A$, $x_0\neq0$, shift $\sigma$, tol. Output: eigenpair near $\sigma$.
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\kill
$x_0=x_0/\|x_0\|_2,\quad \tilde\lambda_0=x_0^TAx_0,\quad i=0$ \\
\texttt{while} $\|Ax_i-\tilde\lambda_i x_i\|>\texttt{tol}$ \\
\> solve $(A-\sigma I)y_{i+1}=x_i$ \\
\> $x_{i+1}=y_{i+1}/\|y_{i+1}\|_2$ \\
\> $\tilde\lambda_{i+1}=x_{i+1}^TAx_{i+1}$ \\
\> $i=i+1$ \\
\texttt{end}
\end{tabbing}
\end{tiny}

\subsection{Orthogonal Iteration}
Generalization of power method for the first $p$ eigenpairs.
Input: $A$, orthonormal $Z_0\in\R^{n\times p}$, tol. Output: first $p$
dominant eigenvalues/eigenspace.
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
$i=0,\quad A_0=Z_0^TAZ_0,\quad L_0=$ lower triangular part of $A_0$ \\
\texttt{while} $\|L_i\|_F/\|A_i\|_F>\texttt{tol}$ \\
\> $Y_{i+1}=AZ_i$ \\
\> $Y_{i+1}=Z_{i+1}R_{i+1}$ \quad (QR, $Z_{i+1}$ orthonormal) \\
\> $A_{i+1}=Z_{i+1}^TAZ_{i+1}$ \\
\> $L_{i+1}=$ lower triangular part of $A_{i+1}$ \\
\> $i=i+1$ \\
\texttt{end}
\end{tabbing}
\end{tiny}
Columns of $Z_i$ converge to the dominant eigenspace; diagonal of $A_i$ gives
the approximated eigenvalues.

\subsection{QR Iteration}
Computes all eigenvalues. Input: $A\in\R^{n\times n}$, tol. Output:
$\lambda_1,\ldots,\lambda_n$.
\begin{tiny}
\begin{tabbing}
\hspace{1.0em}\=\hspace{1.0em}\=\hspace{1.0em}\=\kill
$i=0,\quad A_0=A,\quad L_0=$ lower triangular part of $A_0$ \\
\texttt{while} $\|L_i\|_F/\|A_i\|_F>\texttt{tol}$ \\
\> compute QR factorization $A_i=Q_iR_i$ \\
\> $A_{i+1}=R_iQ_i$ \quad (same eigenvalues as $A_i$) \\
\> $L_{i+1}=$ lower triangular part of $A_{i+1}$ \\
\> $i=i+1$ \\
\texttt{end}
\end{tabbing}
\end{tiny}
The diagonal of $A_i$ converges to the eigenvalues; with shifts it converges
faster.


\subsection{Upper Hessenberg Matrix}
A matrix $H\in\R^{n\times n}$ is \textbf{upper Hessenberg} if
\[
  h_{ij}=0 \quad \text{for } i>j+1.
\]
So everything below the first subdiagonal is zero:
\[
H=
\begin{bmatrix}
* & * & * & \cdots & *\\
* & * & * & \cdots & *\\
0 & * & * & \cdots & *\\
0 & 0 & * & \cdots & *\\
\vdots & \vdots & \ddots & \ddots & *
\end{bmatrix}.
\]
This form is useful because QR iteration preserves it.

\subsection{Reduction to Hessenberg Form}
Before QR iteration, transform
\[
  A \longrightarrow H = Q^T A Q.
\]
This is an orthogonal similarity transformation, so eigenvalues are preserved.

Using Householder reflections, we zero entries below the first subdiagonal,
column by column:
\[
  A_k = Q_k^T A_{k-1} Q_k, \qquad k=1,\ldots,n-2.
\]
At step $k$, take the part of column $k$ below the diagonal:
\[
  x=A_{k-1}(k+1:n,k).
\]
Build a Householder reflector $\widehat H_k$ so
\[
  \widehat H_k x=\pm\|x\|_2 e_1,
\]
therefore entries $k+2,\ldots,n$ in column $k$ become zero. Embed it into the
full matrix:
\[
  Q_k=
  \begin{bmatrix}
    I_k & 0\\
    0 & \widehat H_k
  \end{bmatrix}.
\]
Multiplying by $Q_k^T$ from the left zeros the column; multiplying by $Q_k$ from
the right keeps a similarity transformation, so eigenvalues are unchanged.
Cost: $O(n^3)$, but only once.

\subsection{Givens Rotations}
A Givens rotation zeros one component of a 2D vector:
\[
v=
\begin{bmatrix}
x_1\\x_2
\end{bmatrix},
\qquad
Gv=
\begin{bmatrix}
\sqrt{x_1^2+x_2^2}\\0
\end{bmatrix}.
\]
With
\[
c=\frac{x_1}{\sqrt{x_1^2+x_2^2}},\qquad
s=-\frac{x_2}{\sqrt{x_1^2+x_2^2}},
\]
\[
G=
\begin{bmatrix}
c & -s\\
s & c
\end{bmatrix}.
\]
In large matrices, $G$ only mixes two rows/columns.

\subsection{QR of an Upper Hessenberg Matrix}
For Hessenberg $H$, QR is cheap because every column has only one subdiagonal
entry to eliminate. Use Givens rotations:
\[
  R = G_{n-1}\cdots G_1 H,
\]
where each $G_k$ zeros $h_{k+1,k}$.
At step $k$, build a $2\times2$ rotation from the vector
\[
  x=\begin{bmatrix}h_{k,k}\\ h_{k+1,k}\end{bmatrix},
  \qquad
  \widehat G_kx=
  \begin{bmatrix}*\\0\end{bmatrix}.
\]
Embed it into the full matrix so it only mixes rows $k$ and $k+1$:
\[
  G_k=
  \begin{bmatrix}
    I_{k-1} & 0 & 0\\
    0 & \widehat G_k & 0\\
    0 & 0 & I_{n-k-1}
  \end{bmatrix}.
\]
Apply $G_kH$ to zero the subdiagonal entry. Continue for
$k=1,\ldots,n-1$ to get upper triangular $R$.

Then
\[
  H = QR,
  \qquad
  Q^T = G_{n-1}\cdots G_1.
\]

\subsection{Cost}
Dense QR iteration costs about $O(n^3)$ per step.

For upper Hessenberg matrices:
\[
  \text{QR factorization cost} = O(n^2),
\]
because there are $O(n)$ Givens rotations and each costs $O(n)$.

So the total QR process is reduced roughly from
\[
  O(n^4) \quad \text{to} \quad O(n^3).
\]

\section{Solving Nonlinear Equations}

\subsection{Difference of Function Values}
When $x_a,x_b\to x^*$, we have
\[
  \frac{f(x_b)-f(x_a)}{x_b-x_a}\to f'(x^*)
\]
\[
   |f(x_b)-f(x_a)|\approx |f'(x^*)|\,|x_b-x_a|
\]

\subsection{Bisection}
Finding a zero of function $f(x)$ on interval $[a,b]$, where
$f(a)\cdot f(b)<0$.
At each step take the midpoint and keep the half-interval whose endpoints have
opposite signs. Stop when the interval is shorter than the prescribed tolerance.

\begin{tiny}
\begin{tabbing}
\hspace{1em}\=\hspace{1em}\=\kill
\texttt{for} $k=1,2,\ldots$ \\
\> $x_m=a+(b-a)/2$ \\
\> \texttt{if} $\sgn(f(x_m))=\sgn(f(a))$ \\
\> \> $a=x_m$ \\
\> \texttt{else} \\
\> \> $b=x_m$ \\
\> \texttt{end} \\
\> \texttt{if} $|b-a|<\mathrm{tol}$, \texttt{stop} \\
\texttt{end}
\end{tabbing}
\end{tiny}

After $n$ steps the interval length is
\[
  \delta_n=\frac{b-a}{2^n}.
\]
For accuracy $\varepsilon$, need $\delta_n\leq\varepsilon$:
\[
  n\geq \left\lceil \log_2\frac{b-a}{\varepsilon}\right\rceil.
\]

\subsection{Tangent Method / Newton's Method}
Use the tangent at point $x_k$ to find a zero of $f(x)$:
\[
  x_{k+1}=x_k-\frac{f(x_k)}{f'(x_k)}
\]
Point $x_{k+1}$ is the intersection of the tangent at $x_k$ with the x-axis. The order of convergence is
$r=2$, if $f'(x^*)\neq 0$.

\subsection{Secant Method}
Approximate the derivative using the two previous points, so
we need two initial approximations: $x_0,x_1$:
\[
  f'(x_k)\approx\frac{f(x_k)-f(x_{k-1})}{x_k-x_{k-1}}
\]
\[
  x_{k+1}=x_k-f(x_k)\cdot
  \frac{x_k-x_{k-1}}{f(x_k)-f(x_{k-1})}
\]

\subsection{Fixed Point Method}
Rewrite $f(x)=0$ as an equivalent fixed-point equation
\[
  g(x)=x.
\]
A point $x^*$ with $g(x^*)=x^*$ is a fixed point. Iterate:
\[
  x_{k+1}=g(x_k)
\]
until the tolerance criterion is satisfied.

\subsection{Newton's Method}
Generalization of the tangent method for nonlinear systems
$F(x)=0$.

\[
  J(x_k)\Delta x=-F(x_k)
\]
Solve the linear system for $\Delta x$ by any method, then update:
\[
  x_{k+1}=x_k+\Delta x
\]
\[
  J(x)=
  \begin{bmatrix}
    \frac{\partial f_1}{\partial x_1} & \cdots &
      \frac{\partial f_1}{\partial x_n} \\
    \vdots & \ddots & \vdots \\
    \frac{\partial f_n}{\partial x_1} & \cdots &
      \frac{\partial f_n}{\partial x_n}
  \end{bmatrix}
\]

\subsection{Broyden's Method}
Approximate the Jacobian matrix by $B_r$:
\[
  B_{r+1}=B_r+
  \frac{F(x^{(r+1)})(\Delta x^{(r)})^T}{\|\Delta x^{(r)}\|_2^2}.
\]

\subsection{Variational Methods}
Zeros of $f(x)$ are global minima of the function
\[
  g(x)=\|f(x)\|^2=\sum_{i=1}^n(f_i(x))^2.
\]
Find the minimum iteratively:
\[
  x^{(r+1)}=x^{(r)}+\lambda_r v_r,\qquad
  g(x^{(r+1)})<g(x^{(r)}).
\]

\subsection{Choosing the Descent Direction}
\begin{itemize}
  \item general: $v_r\not\perp\nabla g(x)$,
  \item steepest descent: $v_r=-\nabla g(x)$,
  \item coordinate descent: $v_r=e_1,e_2,\ldots,e_n$ in sequence.
\end{itemize}

\subsection{Gauss--Newton Iteration}
For an overdetermined nonlinear system, use the pseudoinverse of the Jacobian matrix:
\[
  x^{(r+1)}=x^{(r)}-J_f(x^{(r)})^+f(x^{(r)}).
\]

\section{Polynomial Interpolation}

Goal: approximate a function $f(x)$ by an easier function $g(x)$.

Find a polynomial $p(x)$ of degree at most $n-1$ that satisfies:
\[
  p(x_i)=y_i, \quad \text{for } i=1,2,\ldots,n
\]
This means that the polynomial interpolates the given points $(x_i,y_i)$. 

\subsection{Vandermonde System}
This problem can be written as a linear system:
\[
  \begin{bmatrix}
    1 & x_1 & \cdots & x_1^{n-1} \\
    1 & x_2 & \cdots & x_2^{n-1} \\
    \vdots & \vdots & \ddots & \vdots \\
    1 & x_n & \cdots & x_n^{n-1}
  \end{bmatrix}
  \cdot
  \begin{bmatrix}
    a_0 \\ a_1 \\ \vdots \\ a_{n-1}
  \end{bmatrix}
  =
  \begin{bmatrix}
    y_1 \\ y_2 \\ \vdots \\ y_n
  \end{bmatrix}
\]

\subsection{Orthogonal Polynomials}
The Vandermonde system can be ill-conditioned, so it is better to use an orthogonal
basis $g_0,\ldots,g_n$:
\[
  \langle f,g\rangle=\sum_{i=1}^m f(x_i)g(x_i),\qquad
  \langle g_j,g_k\rangle=0\ (j\neq k).
\]
\[
  p(x)=\sum_{i=0}^n a_i g_i(x),\qquad
  a_i=\frac{\langle f,g_i\rangle}{\langle g_i,g_i\rangle}.
\]

\subsection{Lagrange Form}
\[
  \ell_i(x)=\prod_{\substack{j=0\\j\neq i}}^n
  \frac{x-x_j}{x_i-x_j},\qquad i=0,\ldots,n.
\]
\[
  p(x)=\sum_{i=0}^n \ell_i(x)y_i.
\]

\subsection{Newton Form}
With points $x_0,\ldots,x_n$ ($n+1$ points), interpolate by a polynomial of
degree at most $n$.
\[
  p_n(x)=c_0+c_1(x-x_0)+\cdots+
  c_n\prod_{j=0}^{n-1}(x-x_j).
\]
The coefficients are divided differences:
\[
  c_i=f[x_0,\ldots,x_i],\qquad i=0,\ldots,n.
\]
\[
\begin{array}{c|c|c|c}
  x & f[\cdot] & f[\cdot,\cdot] & f[\cdot,\cdot,\cdot] \\ \hline
  x_0 & f[x_0] & & \\
      &        & f[x_0,x_1] & \\
  x_1 & f[x_1] & & f[x_0,x_1,x_2] \\
      &        & f[x_1,x_2] & \\
  x_2 & f[x_2] & & f[x_1,x_2,x_3] \\
      &        & f[x_2,x_3] & \\
  x_3 & f[x_3] & &
\end{array}
\]
Newton coefficients are the top diagonal:
\[
  c_0=f[x_0],\quad c_1=f[x_0,x_1],\quad
  c_2=f[x_0,x_1,x_2],\ldots
\]
\[
  f[x_i]=f(x_i)=y_i,\qquad
  f[x_i,x_j]=\frac{f(x_j)-f(x_i)}{x_j-x_i}.
\]
\[
  f[x_0,\ldots,x_n]=
  \frac{f[x_1,\ldots,x_n]-f[x_0,\ldots,x_{n-1}]}{x_n-x_0}.
\]

\section{Numerical Integration}

\subsection{Gaussian Quadrature Formula}
Goal: approximate an integral with as few function evaluations as possible:
\[
  \int_a^b f(x)w(x)\dd x \approx \sum_{i=1}^N \rho_i f(x_i).
\]
The nodes $x_i$ and weights $\rho_i$ are both chosen. Because there are $2N$
free parameters, the rule can be exact for all polynomials up to degree $2N-1$.

\textbf{General construction:}
\begin{enumerate}
  \item Choose the number of nodes $N$.
  \item Find the degree $N$ orthogonal polynomial $q_N$ for
  \[
    \langle f,g\rangle_w=\int_a^b f(x)g(x)w(x)\dd x.
  \]
  Equivalently, $q_N$ satisfies
  \[
    \int_a^b x^k q_N(x)w(x)\dd x=0,\qquad k=0,\ldots,N-1.
  \]
  \item The nodes are the roots of this polynomial:
  \[
    q_N(x_i)=0,\qquad i=1,\ldots,N.
  \]
  \item Find weights from exactness on $1,x,\ldots,x^{N-1}$:
  \[
    \sum_{i=1}^N \rho_i x_i^k
    =\int_a^b x^k w(x)\dd x,\qquad k=0,\ldots,N-1.
  \]
\end{enumerate}

For $w=1$ on $[-1,1]$, $q_N$ are Legendre polynomials.

\textbf{Move interval $[a,b]$ to $[-1,1]$:}
Use the linear substitution
\[
  t=a_0+a_1x,\qquad t(a)=-1,\quad t(b)=1.
\]
Then
\[
  a_0=-\frac{b+a}{b-a},\qquad a_1=\frac{2}{b-a}.
\]
Equivalently, solve for $x$ in terms of $t$:
\[
  x=\frac{b-a}{2}t+\frac{b+a}{2},
  \qquad
  dx=\frac{b-a}{2}\dd t.
\]
Therefore
\[
  \int_a^b f(x)\dd x
  =
  \int_{-1}^1
  f\left(\frac{(b-a)t+b+a}{2}\right)
  \frac{b-a}{2}\dd t.
\]

\textbf{Absolute vs. relative error:}
If $I$ is the exact integral and $\hat I$ is the numerical approximation, then
\[
  E_{\mathrm{abs}}=|I-\hat I|,
  \qquad
  E_{\mathrm{rel}}=\frac{|I-\hat I|}{|I|}.
\]
Absolute error tells the size of the mistake in the original units. Relative
error tells how large the mistake is compared with the true value.

\subsection{Newton--Cotes Rules}
A special form of least squares, where nodes $x_i$ are chosen
equidistantly on interval $[a,b]$:
\[
  x_i=a+ih, \quad h=\frac{b-a}{mn}, \quad i=0,\ldots,n
\]
\textbf{Trapezoidal Rule} ($n=1$, $\leq 1$):
\[
T  =\frac{h}{2}(f(a)+f(b))-\frac{h^3 f''(\xi)}{12}
\]
\textbf{Simpson's Rule} ($n=2$, $\leq 3!$):
\[
S  =\frac{h}{3}(f(x_0)+4f(x_1)+f(x_2))-\frac{h^5 f^{(4)}(\xi)}{90}
\]
\textbf{Composite Trapezoidal} ($m>1$):
\[
CT  =\frac{h}{2}(f(x_0)+2f(x_1)+\cdots+2f(x_{n-1})+
\]
\[
  f(x_n))-\frac{(b-a)h^2f''(\xi)}{12}
\]
\textbf{Composite Simpson} ($m>1$):
\[
CS  =\frac{h}{3}(f(x_0)+4f(x_1)+2f(x_2)+\cdots+
\]
\[
  2f(x_{n-2})+4f(x_{n-1})+f(x_n))-
  \frac{(b-a)h^4f^{(4)}(\xi)}{180}
\]

\subsection{Romberg Method}
$h=(b-a)/n$ is the subinterval width of the composite trapezoidal rule $T(h)$.
The error has an expansion in even powers:
\[
  T(h)=I+c_1h^2+c_2h^4+\cdots
\]
In each new row, double the number of subintervals, so
$h_\ell=h_0/2^\ell$. Start with
\[
  R_{\ell,0}=T(h_\ell)
\]
and compute the triangular table:
\[
  R_{\ell,m}=
  \frac{4^mR_{\ell,m-1}-R_{\ell-1,m-1}}{4^m-1},
  \qquad 1\leq m\leq\ell.
\]
\[
\begin{array}{c|c|c|c}
  h & R_{\ell,0}=T(h_\ell) & R_{\ell,1} & R_{\ell,2} \\ \hline
  h_0     & R_{0,0} & & \\
  h_0/2   & R_{1,0} & R_{1,1} & \\
  h_0/4   & R_{2,0} & R_{2,1} & R_{2,2} \\
  h_0/8   & R_{3,0} & R_{3,1} & R_{3,2}
\end{array}
\]
Best approximations are on the top diagonal:
\[
  R_{0,0},\quad R_{1,1},\quad R_{2,2},\ldots
\]
Each new column eliminates the next error term: $h^2,h^4,\ldots$


\section{Differential Equations (DE)}

\subsection{Definition.}
DE: equation of the form:
\[
  F(x,y,y',y'',\ldots,y^{(n)})=0, \quad y=y(x)
\]
The general solution is a family of functions with $n$ parameters. A particular solution is
obtained when boundary conditions are given.

Order of a DE: the highest derivative that appears.

\subsection{Introducing a New Variable.}
Convert a higher-order ODE to a first-order system by naming each derivative:
\[
  u_1=y,\quad u_2=y',\quad \ldots,\quad u_n=y^{(n-1)}.
\]
Then
\[
  u_1'=u_2,\quad u_2'=u_3,\quad \ldots,\quad
  u_n'=F(x,u_1,\ldots,u_n).
\]
Example:
\[
  y''=F(x,y,y') \Rightarrow
  \begin{cases}
    u_1'=u_2,\\
    u_2'=F(x,u_1,u_2).
  \end{cases}
\]
For $y''=y'+2y$:
\[
  u_1=y,\quad u_2=y',\quad
  u_1'=u_2,\quad u_2'=u_2+2u_1.
\]

\subsection{Separable First-Order DE}
Recognize equations where all $x$ terms can go on one side and all $t$ terms
on the other:
\[
  \dot x=f(t)g(x),\qquad \dot x=\frac{dx}{dt}.
\]
Procedure:
\[
  \frac{dx}{dt}=f(t)g(x)
  \Rightarrow
  \frac{dx}{g(x)}=f(t)\dd t
\]
\[
  \Rightarrow\quad
  \int\frac{1}{g(x)}\dd x=\int f(t)\dd t+C.
\]
Then solve the obtained equation for $x(t)$ if possible.

\subsection{Linear First-Order DE}
Standard form:
\[
  y'+f(x)y=g(x).
\]
Use the integrating factor
\[
  \mu(x)=e^{\int f(x)\dd x}.
\]
Then multiply the whole equation by $\mu$:
\[
  \mu y'+\mu f y=\mu g
  \quad\Rightarrow\quad
  (\mu y)'=\mu g.
\]
Integrate and divide by $\mu$:
\[
  \mu y=\int \mu(x)g(x)\dd x+C,
  \qquad
  y=\frac{1}{\mu}\left(\int \mu g\dd x+C\right).
\]
\subsection{Local and Global Error.}
\textbf{Local error:} the difference between approximation $y_i$, obtained from approximation $y_{i-1}$,
and $z(x_i)$, the result if the same step were taken from exact
$y(x_{i-1})$.
\[
  l_i=y_i-z(x_i)
\]
\textbf{Global error:} the difference between approximation $y_i$ and exact $y(x_i)$ at $x_i$:
\[
  g_i=y_i-y(x_i) \leq \sum |l_j|
\]
\subsection{Order of a Method}
$p \in \mathbb{N}$, for which:
\[
  l_i=Ch^{p+1}+O(h^{p+2})
\]
The order of a method determines the accuracy of the approximation (higher $p$ $\to$ better method).

\subsection{Euler's Method}
\textbf{Order 1:} linear approx. using the tangent at $x_i$.

\textbf{Explicit:}
\[
  y_{i+1}=y_i+h_i\cdot f(x_i,y_i)
\]
\textbf{Implicit:}
\[
  y_{i+1}=y_i+h_i\cdot f(x_{i+1},y_{i+1})
\]

\subsection{Runge--Kutta 4 (RK4)}
For the IVP $y'=f(x,y)$, $y(x_0)=y_0$ and step $h$:
\[
  k_1=f(x_n,y_n)
\]
\[
  k_2=f\left(x_n+\frac{h}{2},y_n+\frac{h}{2}k_1\right)
\]
\[
  k_3=f\left(x_n+\frac{h}{2},y_n+\frac{h}{2}k_2\right)
\]
\[
  k_4=f(x_n+h,y_n+hk_3)
\]
\[
  y_{n+1}=y_n+\frac{h}{6}(k_1+2k_2+2k_3+k_4)
\]
Idea: estimate the slope at start, two midpoints, and end;
then use the weighted average $1:2:2:1$.
\[
  x_{n+1}=x_n+h
\]
\textbf{Order 4:} local error $O(h^5)$, global error $O(h^4)$.
For systems, use the same formulas with vector $y_n$ and slopes $k_i$.


\section{Appendix}

\subsection{Line Equation}
Through two points $(x_0,y_0)$ and $(x_1,y_1)$:
\[
  y-y_0=\frac{y_1-y_0}{x_1-x_0}(x-x_0).
\]
Slope-intercept form:
\[
  y=kx+n,\qquad k=\frac{y_1-y_0}{x_1-x_0}.
\]

\subsection{Taylor Series}
Replace a function by a polynomial near point $a$:
\[
  \begin{aligned}
    f(x)={}&f(a)+f'(a)(x-a)\\
    &+\frac{f''(a)}{2!}(x-a)^2+
    \frac{f'''(a)}{3!}(x-a)^3+\cdots
  \end{aligned}
\]
General form:
\[
  f(x)=\sum_{k=0}^{\infty}\frac{f^{(k)}(a)}{k!}(x-a)^k.
\]
